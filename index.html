async function recognizeAndAttack() { if (inkData.length === 0) return; document.getElementById('loading').style.display = 'flex'; try { const res = await fetch(`https://inputtools.google.com/request?itc=ja-t-i0-handwrit&app=test`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ app_version: 0.4, requests: [{ ink: inkData, language: "ja" }] }) }); const data = await res.json(); const result = data[1][0][1]; // Googleからの予測候補リスト let damage = 0; let message = ""; if (result && result[0] === currentMondai.kanji) { // 【超成功】1番目の候補が正解と一致 damage = 50; message = "会心の漢術！"; } else if (result && result.slice(1, 3).includes(currentMondai.kanji)) { // 【惜しい】2〜3番目に入っている damage = 15; message = "形が乱れているぞ！"; } if (damage > 0) { // 成功エフェクト（ダメージが大きいほど激しく） flashOverlay.style.backgroundColor = (damage > 30) ? "white" : "rgba(255,255,255,0.5)"; flashOverlay.classList.add('flash'); enemyGraphic.classList.add('shake'); enemyHp -= damage; enemyHpBar.style.width = Math.max(0, enemyHp) + '%'; setTimeout(() => { flashOverlay.classList.remove('flash'); enemyGraphic.classList.remove('shake'); if (enemyHp <= 0) { alert("魔物を完全に封印した！"); enemyHp = 100; enemyHpBar.style.width = '100%'; } else { // ダメージが低い時は同じ問題を続行させるのもアリですが、 // ストレス軽減のため次へ行く仕様にしています console.log(message); } next(); }, 400); } else { // 【失敗】候補のトップ3にすらいない alert(`鑑定不可：これは「${result[0]}」ではないか？\n正解は「${currentMondai.kanji}」だ！`); clearCanvas(); } } catch(e) { alert("通信エラー"); } document.getElementById('loading').style.display = 'none'; }
